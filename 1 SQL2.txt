                      SalesAndRevenueYear AS [Year], SUBSTRING(DATENAME(mm, CAST(SalesAndRevenueMonth AS VARCHAR)

                      SalesAndRevenueMonth

                          (SELECT     FiscalYear, a.ProjectCode, b.SalesAndRevenueMonth, b.SalesAndRevenueYear,

                                                         WHERE      SalesAndRevenueCode = b.SalesAndRevenueCode) AS Sales,
                                                       (SELECT     ISNULL(BacklogRevenueAmount, 0)
                                                         FROM          tblSalesAndRevenue
                                                         WHERE      SalesAndRevenueCode = b.SalesAndRevenueCode) AS Backlog,
                                                       (SELECT     ISNULL(ShadowRevenueAmount, 0)
                                                         FROM          tblSalesAndRevenue
                                                         WHERE      SalesAndRevenueCode = b.SalesAndRevenueCode) AS Shadow
                            FROM          tlkpProject a, tblSalesAndRevenue b
                            WHERE      a.ProjectCode = b.ProjectCode) b,
                          (SELECT     ClientCode, ClientDecode, ClientGrpDecode,
                                                       (SELECT     UserName
                                                         FROM          tlkpUser
                                                         WHERE      UserId = a.ClientSrExec) AS ClientSrExecName
                            FROM          tlkpClient a, tlkpClientGroup b
                            WHERE      a.ClientGrpCode = b.ClientGrpCode) c, tblClientRevenue d

GROUP BY ClientGrpDecode, ClientDecode, ClientSrExecName, SalesAndRevenueYear, SalesAndRevenueMonth
